// Called on every completed WRITE: arm a full physical re-commit and (re)start
// the idle timer. lo/hi are unused now (kept for call-site compatibility) - the
// whole physical region is rewritten regardless of which logical bytes changed,
// because logical->physical is not a usable mapping here. The cursor is reset to
// "not started" so that a write arriving MID-DRAIN abandons the partial pass and
// restarts from block 0 after the new idle period.
static inline void ost_persist_mark(uint32_t lo, uint32_t hi) {
    (void)lo; (void)hi;
    ost_persist_dirty    = true;
    ost_persist_phys_off = 0xFFFFFFFFu;       // restart any in-progress drain
    ost_persist_last_ms  = board_millis();
}

// Re-arm the idle debounce WITHOUT marking dirty or disturbing an in-progress
// drain's dirty flag. Called on every write byte so that a long multi-command
// upload (e.g. TunerPro streaming a 64 KB image as many 'W' commands) keeps the
// commit suppressed until the host has been fully idle for OST_PERSIST_IDLE_MS.
static inline void ost_persist_touch(void) {
    ost_persist_last_ms  = board_millis();
    ost_persist_phys_off = 0xFFFFFFFFu;       // abandon any partial drain
}

// Called from the main loop. Once writes have been idle long enough, drain the
// physical image in 4 KB blocks ONE PER CALL: each commit is a short
// interrupts-off window for a single physical block, and the main loop
// (tud_task) runs between blocks so USB stays responsive.
static void ost_persist_task(void) {
    if (!ost_persist_dirty) return;
    // Never commit while a WRITE is in flight
    if (ost.state == OST_WRITE_DATA || ost.state == OST_WRITE_PARAMS) return;
    // Never begin/continue a commit while bytes are waiting on the Ostrich CDC
    if (tud_cdc_n_available(OST_CDC_ITF) > 0) { ost_persist_touch(); return; }
    if ((uint32_t)(board_millis() - ost_persist_last_ms) < OST_PERSIST_IDLE_MS) return;
    if (ost_persist_phys_off == 0xFFFFFFFFu) ost_persist_phys_off = 0u;  // start drain
    bool more = ost_persist_commit();         // one physical block
    if (!more) {                              // whole physical image flushed
        ost_persist_dirty    = false;
        ost_persist_phys_off = 0xFFFFFFFFu;
    }
    // if more: leave dirty set, timer un-advanced -> next iteration drains next block
}
